currency (货币处理)

before: {{ price }}

after: {{ price | currency : '￥' }}

date (日期格式化)

before: {{ time }}

after: {{time | date : 'yyyy-MM-dd hh:mm:ss EEEE'}}

filter (从数组项中选择一个子集)

before: {{ childrenArray }}

after: {{ childrenArray | filter : 'a' }} //匹配属性值中含有a的

after: {{ childrenArray | filter : 4 }} //匹配属性值中含有4的

after: {{ childrenArray | filter : {name : 'i'} }} //参数是对象，匹配name属性中含有i的

json (格式化json对象)

before: {{ childrenObject }}

after: {{ childrenObject | json }}

limitTo(限制数组长度或字符串长度)

before: {{ childrenArray }}

after: {{ childrenArray | limitTo : 3 }} //限制长度3

before: {{ childrenString }}

after: {{ childrenString | limitTo : 3 }} //限制长度3

lowercase(小写)

before: {{ childrenString }}

after: {{ childrenString | lowercase }}

uppercase(大写)

before: {{ childrenString }}

after: {{ childrenString | uppercase }}

number(格式化数字)

before: {{ time }}

after: {{ time | number : 2}} //保留两位小数

orderBy(排序)

before: {{ childrenArray }}

after: {{ childrenArray | orderBy : ['-age','name'] : true }}
//用法：orderBy:predicate:reverse
//age前面的'-'表示age这列倒序排序，默认为正序排序
//reverse：true表示结果集倒序显示，默认为正序排序

after: {{ childrenArray | filter : {name : 'i'} | orderBy : '-age' }}
//参数是对象，匹配name属性中含有i的，然后按照年龄从大到小排序
